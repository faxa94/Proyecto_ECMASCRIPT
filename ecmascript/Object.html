Funciones

Se puede utilizar los elementos que componen un objeto, si la funcion retorna un objeto. Se especifica el nombre de un elemento perteneciente al objeto para obtener el valor.

            
function persona(nombre, apellido){
let completo= nombre + " " + apellido;
return { nombre: nombre, apellido: apellido, completo: completo }
}
let n1 = persona("Teresa", "Paz");
console.log( n1.nombre ); //Teresa
console.log( n1.apellido );//Paz
console.log( n1.completo);// Teresa Paz

Tambien se puede utilizar su forma corta para el retorno del objeto

            
function persona(nombre, apellido){
let completo= nombre + " " + apellido;
return { nombre, apellido, completo }
}
let n1 = persona("Teresa", "Paz");
console.log( n1.nombre ); //Teresa
console.log( n1.apellido );//Paz
console.log( n1.completo);// Teresa Paz

Inicializacion de Objetos

Se puede utilizar la forma corta para la declaracion e inicialicion de los valores de un objeto.

  
  let nombre= "Teresa";
  let apellido= "Paz";
  let completo= nombre + " " + apellido;
let n1 = {nombre,apellido,completo};
console.log( n1.nombre ); //Teresa
console.log( n1.apellido );//Paz
console.log( n1.completo);// Teresa Paz

Object Destructuring

Permite la extracción de datos de objetos usando una sintaxis que equivale a la construcción de objetos literales.

  
//Forma tradicional
function persona(nombre, apellido){
let completo= nombre + " " + apellido;
return { nombre, apellido, completo }
}
let n1 = persona("Teresa", "Paz");
let nombre= n1.nombre ;
let apellido= n1.apellido;
let completo= n1.completo;
  

  
//Forma Destructuring
function persona(nombre, apellido){
let completo= nombre + " " + apellido;
return { nombre, apellido, completo }
}
let { nombre, apellido, completo } = persona("Teresa", "Paz");
console.log(nombre,apellido,completo);
