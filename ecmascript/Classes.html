Clases en ES6

Una clase se reconoce porque utiliza la palabra reservada 'class'. Se define un constructor y los metodos adicionales que son propios de la funcion.

     
  class Personal {
        constructor(ci, nombre, apellido) {
            this.ci = ci;
            this.nombre= nombre;
            this.apellido= apellido;
        }
        toString() {
            return this.ci + ', ' + this.nombre+ ' ' +this.apellido;
        }
    }
    
   
 let cp = new Personal(18,'andres', 'reyes');
console.log(cp.toString());

Se puede instanciar una clase o hacer uso de ella a traves de la herencia. Para este proposito se declara la palabra reservada extends

     
           class Personal {
        constructor(ci, nombre, apellido) {
            this.ci = ci;
            this.nombre= nombre;
            this.apellido= apellido;
        }
        toString() {
            return this.ci + ', ' + this.nombre+ ' ' +this.apellido;
        }
    }
    class Registro extends Personal{
       constructor(ci, nombre, apellido, genero){
         super(ci, nombre, apellido);
         this.genero= genero;
       }
        toString() {
            return super.toString() + ' + ' + this.genero;
        }
    }
 let cp = new Registro(18,'andres', 'reyes', 'masculino');
console.log(cp.toString());
