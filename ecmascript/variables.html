Variables: VAR, LET y CONST


LET

Nueva variable de ES6, su declaracion es igual que la variable var utilizada en ES5.

            
              let messajes= "variable global";

La variable let tiene un alcance limitado, es decir esta activa solamente dentro de su scope

            
let messajes= "variable global";
function fn1(){
  let messajes="variable local"
  console.log(messajes);
  return messajes;
}

Resuelve problemas de closures ocasionados durante la utilizacion de estructuras de datos de repeticion.

        
   var arreglo = ['a','b','c','d'];
function encuentraPosicionLetra(letra, arreglo){
  var devolverValor;
  for(let i=1;i<=arreglo.length;i++){
    if(letra==arreglo[i]){
      devolverValor = function(){
        console.log('La posicion es: '+i);
      }
    }
  }
  return devolverValor;
}
var respuesta = encuentraPosicionLetra('b',arreglo)
respuesta(); // retorna el valor 1
CONST

Se utiliza en valores que no pueden ser modificados en todo el programa. Su declaracion e inicializacion se realiza de la siguiente forma:

  
    const pi=3.14;

Su valor no se desplega si existe modificaciones en su valor

  
    const pi=3.14;
function encuentraPosicionLetra(letra, arreglo){
  var devolverValor;
  
  for(let i=0;i<=arreglo.length;i++){
     pi=5654; //modificar valor
    if(letra==arreglo[i]){
      devolverValor = function(){
        console.log('La posicion es: '+pi); //error 
      }
    }
  }
  return devolverValor;

}
